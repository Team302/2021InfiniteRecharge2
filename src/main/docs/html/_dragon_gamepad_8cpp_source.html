136 double DragonGamepad::GetAxisValue
137 (
138  AXIS_IDENTIFIER axis
139 ) const
140 {
141  double value = 0.0;
142  if ( m_axis[axis] != nullptr )
143  {
144  value = m_axis[axis]->GetAxisValue();
145  if ( axis == AXIS_IDENTIFIER::GAMEPAD_AXIS_16 || axis == AXIS_IDENTIFIER::GAMEPAD_AXIS_17 )
146  {
147  value -= JOYSTICK_OFFSET;
148  }
149  }
150  else
151  {
152  Logger::GetLogger()->LogError( string("DragonGamepad::GetAxisValue no axis"), to_string(axis) );
153  }
154  return value;
