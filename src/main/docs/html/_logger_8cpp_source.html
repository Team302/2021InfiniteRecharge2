94 void Logger::LogError
95 (
96  LOGGER_LEVEL level,
97  const string& locationIdentifier,
98  const string& message
99 )
100 {
101  if ( level <= m_level )
102  {
103  auto display = true;
104  if ( level%2 == 1 )
105  {
106  string key = locationIdentifier + message;
107  auto it = m_alreadyDisplayed.find(key);
108  display = it == m_alreadyDisplayed.end();
109  if (display)
110  {
111  m_alreadyDisplayed.insert(key);
112  }
113 
114  }
115  if (display)
116  {
117  switch ( m_option )
118  {
119  case LOGGER_OPTION::CONSOLE:
120  cout << locationIdentifier << ": " << message << endl;
121  break;
122 
123  case LOGGER_OPTION::DASHBOARD:
124  SmartDashboard::PutString( locationIdentifier.c_str(), message.c_str());
125  break;
126 
127  default: // case LOGGER_OPTION::EAT_IT:
128  break;
129 
130  }
131  }
132  }
133 }
