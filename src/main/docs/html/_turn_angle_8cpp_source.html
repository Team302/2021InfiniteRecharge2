90 void TurnAngle::Init
91 (
92  PrimitiveParams* params
93 )
94 {
95  m_isDone = false;
96 
97  // get the current angle (heading) of the robot
98  auto pigeon = PigeonFactory::GetFactory()->GetPigeon();
99  auto startHeading = pigeon != nullptr ? pigeon->GetYaw() : 0.0;
100  auto angle = params->GetHeading();
101 
102  // if relative, add the angle to the current heading to get a target
103  // if absolute, the target angle is what was specified
104  auto isRelative = params->GetID() == TURN_ANGLE_REL;
105  m_targetAngle = isRelative ? (startHeading + angle) : angle;
106  auto delta = isRelative ? angle : (startHeading - m_targetAngle);
107  m_turnRight = delta > 0.0;
108 
109  // figure out how far each side of the robot needs to move to turn in place (circle/arc) in order to reach
110  // the angle. So, 2 * PI * radius is the distance a wheel would travel to turn 360 degrees.
111  // Thus, to travel 90 degrees, the wheel would need to move a quarter of this distance (90/360).
112  // This won't be entirely accurate because of scrub and slippage, the encoder may show a slightly different distance
113  // but we'll use this to get approximately there and then use the gyro and a slow speed to finish it up.
114  auto turningRadius = m_chassis->GetTrack() / 2.0;
115  auto arcLen = ((2.0 * M_PI * turningRadius) * delta) / 360.0;
116  m_leftPos = m_chassis->GetCurrentLeftPosition() + arcLen;
117  m_rightPos = m_chassis->GetCurrentRightPosition() - arcLen;
118 
119  // Set the output to the wheels
120  m_chassis->SetControlConstants( m_control );
121  m_chassis->SetOutput( m_control->GetMode(), m_leftPos, m_rightPos );
122 
123  // Get the timeout and start a timer (we may hit something that prevents us from getting to the angle)
124  m_maxTime = params->GetTime();
125  m_timer->Reset();
126  m_timer->Start();
127 }

131 void TurnAngle::Run()
132 {
133  // Check the pigeon to see if we've reached the target, if we have set m_isDone to true and the speeds to 0.0
134  auto pigeon = PigeonFactory::GetFactory()->GetPigeon();
135  auto heading = ( pigeon != nullptr ) ? pigeon->GetYaw() : 0.0;
136 
137  auto delta = heading - m_targetAngle;
138  m_isDone = ( abs(delta) < ANGLE_THRESH );
139  if ( !m_isDone )
140  {
141  auto currLeftPos = m_chassis->GetCurrentLeftPosition();
142  auto currRightPos = m_chassis->GetCurrentRightPosition();
143 
144  auto distReached = (( abs(currLeftPos - m_leftPos) < ANGLE_THRESH ) && (abs(currRightPos - m_rightPos) < ANGLE_THRESH ));
145 
146  // if the distance has been reached but the angle has been reach (only way we got here)
147  // use a slow speed (percent output), to get the last little bit
148  if ( distReached )
149  {
150  // if the target angle hasn't been reached, make the robot hasn't turned too far, if it has turn the opposite way
151  double leftspeed = 0.0;
152  double rightspeed = 0.0;
153  auto passedTarget = ( delta > 0.0 ) ? m_turnRight : !m_turnRight;
154 
155  if ( m_turnRight && !passedTarget )
156  {
157  leftspeed = passedTarget ? -1.0 * SLOW_PERCENT : SLOW_PERCENT;
158  rightspeed = passedTarget ? SLOW_PERCENT : -1.0 * SLOW_PERCENT;
159  }
160  else
161  {
162  leftspeed = passedTarget ? SLOW_PERCENT : -1.0 * SLOW_PERCENT;
163  rightspeed = passedTarget ? -1.0 * SLOW_PERCENT : SLOW_PERCENT;
164  }
165  m_chassis->SetOutput( ControlModes::PERCENT_OUTPUT, leftspeed, rightspeed );
166  }
167  }
168  else
169  {
170  m_chassis->SetOutput( ControlModes::PERCENT_OUTPUT, 0.0, 0.0 );
171  }
172 }
