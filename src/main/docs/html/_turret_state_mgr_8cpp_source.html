40 TurretStateMgr::TurretStateMgr() : m_stateVector(),
41  m_currentState(),
42  m_approxTargetAngle( 0.0 )
43 {
44  auto turret = MechanismFactory::GetMechanismFactory()->GetIMechanism( MechanismTypes::TURRET );
45  m_approxTargetAngle = turret->GetCurrentPosition();
46 
47  // Parse the configuration file
48  auto stateXML = make_unique<StateDataDefn>();
49  vector<MechanismTargetData*> targetData = stateXML.get()->ParseXML( MechanismTypes::MECHANISM_TYPE::TURRET );
50 
51  map<string, TURRET_STATE> stateMap;
52  stateMap["TURRETHOLD"] = TURRET_STATE::HOLD;
53  stateMap["TURRETAUTOAIM"] = TURRET_STATE::LIMELIGHT_AIM;
54  stateMap["TURRETMANUALAIM"] = TURRET_STATE::MANUAL_AIM;
55  stateMap["TURRETTURNANGLE"] = TURRET_STATE::TURN_ANGLE;
56  m_stateVector.resize(4);
57 
58  for ( auto td: targetData )
59  {
60  auto stateString = td->GetStateString();
61  auto stateItr = stateMap.find( stateString );
62  if ( stateItr != stateMap.end() )
63  {
64  auto stateEnum = stateItr->second;
65  if ( m_stateVector[stateEnum] == nullptr )
66  {
67  auto controlData = td->GetController();
68  auto target = td->GetTarget();
69  switch ( stateEnum )
70  {
71  case TURRET_STATE::HOLD:
72  {
73  auto thisState = new HoldTurretPosition(controlData, m_approxTargetAngle, MechanismTargetData::SOLENOID::NONE);
74  m_stateVector[stateEnum] = thisState;
75  m_currentState = thisState;
76  m_currentStateEnum = stateEnum;
77  m_currentState->Init();
78  }
79  break;
80 
81  case TURRET_STATE::LIMELIGHT_AIM:
82  {
83  auto thisState = new LimelightAim(controlData, target);
84  m_stateVector[stateEnum] = thisState;
85  }
86  break;
87 
88  case TURRET_STATE::MANUAL_AIM:
89  {
90  auto thisState = new ManualAim(controlData);
91  m_stateVector[stateEnum] = thisState;
92  }
93  break;
94 
95  case TURRET_STATE::TURN_ANGLE:
96  {
97  auto thisState = new TurretTurnAngle(controlData, td->GetTarget(), MechanismTargetData::SOLENOID::NONE);
98  m_stateVector[stateEnum] = thisState;
99  }
100  break;
101 
102  default:
103  {
104  Logger::GetLogger()->LogError( string("TurretHoodStateMgr::TurretHoodStateMgr"), string("unknown state"));
105  }
106  break;
107  }
108  }
109  {

111  }

113  }

117 void TurretStateMgr::RunCurrentState()
118 {
119  if ( MechanismFactory::GetMechanismFactory()->GetIMechanism( MechanismTypes::MECHANISM_TYPE::TURRET) != nullptr )
120  {
121  // process teleop/manual interrupts
122  auto controller = TeleopControl::GetInstance();
123  if ( controller != nullptr )
124  {
125  if (controller->IsButtonPressed( TeleopControl::FUNCTION_IDENTIFIER::TURRET_MANUAL_BUTTON))
126  {
127  SetCurrentState( TURRET_STATE::MANUAL_AIM, false , 0.0);
128  }
129  if (controller->IsButtonPressed( TeleopControl::FUNCTION_IDENTIFIER::TURRET_LIMELIGHT_AIM))
130  {
131  SetCurrentState( TURRET_STATE::LIMELIGHT_AIM, false, 0.0);
132  }
133  }
134  Logger::GetLogger()->OnDash(string("Turret State"), to_string(m_currentStateEnum));
135  if ( m_currentState != nullptr )
136  {
137  m_currentState->Run();
138  }
139  }
